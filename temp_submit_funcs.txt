        async function submitReadingAnswer() {
            const challenge = readingState.currentChallenge;
            let finalPayload = {};
            
            if (challenge.multi_questions && challenge.multi_questions.length > 0) {
                const answersList = [];
                let allAnswered = true;
                
                challenge.multi_questions.forEach((q, idx) => {
                    const ans = document.getElementById(`readingAnswer_${idx}`)?.value || '';
                    if (!ans.trim()) allAnswered = false;
                    answersList.push({
                        id: q.id,
                        question: q.question,
                        answer: ans,
                        example_answer: q.example_answer
                    });
                });
                
                if (!allAnswered) {
                    showToast('warning', 'Incomplete', 'Please answer all questions before submitting');
                    return;
                }
                
                finalPayload = {
                    answers: answersList,
                    passage: challenge.content || challenge.passage
                };
            } else {
                const answer = document.getElementById('readingAnswer')?.value || '';
                if (!answer.trim()) {
                    showToast('warning', 'Empty Answer', 'Please write an answer before submitting');
                    return;
                }
                
                finalPayload = {
                    answer: answer,
                    passage: challenge.content || challenge.passage,
                    question: challenge.instructions || 'What is the main idea of this passage?',
                    example_answer: challenge.example_answer || ''
                };
            }

            try {
                showToast('info', 'Analyzing', '⏳ AI is analyzing your answers...');
                
                // Disable all textareas and button
                const textareas = document.querySelectorAll('.reading-answer-box, #readingAnswer');
                textareas.forEach(ta => ta.disabled = true);
                if (document.getElementById('readingSubmitBtn')) document.getElementById('readingSubmitBtn').disabled = true;

                // Stop timer
                if (readingTimerInterval) {
                    clearInterval(readingTimerInterval);
                    readingTimerInterval = null;
                }

                // Send to backend for evaluation
                const response = await fetch(`${API_BASE_URL}/stage1/evaluate-reading-answer`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Authorization': `Bearer ${localStorage.getItem('access_token')}`
                    },
                    body: JSON.stringify(finalPayload)
                });

                if (response.ok) {
                    const result = await response.json();
                    console.log('✅ Evaluation result:', result);

                    // Show feedback
                    document.getElementById('readingChallengeContent').style.display = 'none';

                    const feedbackDiv = document.getElementById('readingFeedback');
                    const feedbackTitle = document.getElementById('readingFeedbackTitle');
                    const feedbackMessage = document.getElementById('readingFeedbackMessage');

                    const perfection = result.perfection || result.accuracy || result.score || 0;
                    const creditsEarned = Math.max(5, Math.round((perfection / 100) * (readingState.currentChallenge.credits || 10)));

                    feedbackTitle.innerHTML = `<i class="fas fa-book-reader" style="color: var(--primary);"></i> Reading Analysis Complete!`;

                    let feedbackHTML = `
                        <div style="background: rgba(30, 41, 59, 0.5); padding: 25px; border-radius: 15px; margin-top: 20px; border: 1px solid var(--glass-border);">
                            <div style="display: flex; justify-content: space-around; margin-bottom: 25px; text-align: center;">
                                <div>
                                    <div style="font-size: 24px; font-weight: 700; color: var(--primary);">${perfection}%</div>
                                    <div style="font-size: 12px; color: var(--text-muted);">ACCURACY</div>
                                </div>
                                <div style="border-left: 1px solid var(--glass-border); padding-left: 20px;">
                                    <div style="font-size: 24px; font-weight: 700; color: #10b981;">+${creditsEarned}</div>
                                    <div style="font-size: 12px; color: var(--text-muted);">CREDITS</div>
                                </div>
                            </div>
                            
                            <div style="margin-bottom: 20px;">
                                <h4 style="color: var(--text); margin-bottom: 8px;"><i class="fas fa-info-circle" style="color: var(--primary);"></i> Teacher's Feedback:</h4>
                                <p style="color: var(--text); font-size: 14px; line-height: 1.6; background: rgba(99, 102, 241, 0.05); padding: 15px; border-radius: 10px; border-left: 3px solid var(--primary);">
                                    ${result.feedback || 'Good comprehension! Keep reading to improve further.'}
                                </p>
                            </div>

                            ${result.tamil_feedback ? `
                            <div style="margin-bottom: 20px;">
                                <h4 style="color: var(--text); margin-bottom: 8px;"><i class="fas fa-language" style="color: #ed4fa1;"></i> Tamil Explanation:</h4>
                                <p style="color: var(--text); font-size: 14px; line-height: 1.6; background: rgba(237, 79, 161, 0.05); padding: 15px; border-radius: 10px; border-left: 3px solid #ed4fa1;">
                                    ${result.tamil_feedback}
                                </p>
                            </div>
                            ` : ''}

                            ${result.detailed_analysis ? `
                            <div style="margin-bottom: 20px;">
                                <h4 style="color: var(--text); margin-bottom: 8px;"><i class="fas fa-list-check" style="color: #10b981;"></i> Detailed Analysis:</h4>
                                <p style="color: var(--text-muted); font-size: 13px; line-height: 1.6; background: rgba(16, 185, 129, 0.05); padding: 15px; border-radius: 10px;">
                                    ${result.detailed_analysis}
                                </p>
                            </div>
                            ` : ''}

                            <div style="margin-top: 10px;">
                                <button class="btn-secondary" onclick="backToReadingMenu()" style="width: 100%;">
                                    Try Another Challenge
                                </button>
                            </div>
                        </div>
                    `;

                    feedbackMessage.innerHTML = feedbackHTML;
                    feedbackDiv.style.display = 'block';
                    console.log('✅ Reading evaluation successful:', result);
                } else {
                    const errorText = await response.text();
                    console.error('❌ Evaluation error:', response.status, errorText);
                    showToast('error', 'Evaluation Error', 'Could not evaluate your answer');
                }
            } catch (error) {
                console.error('❌ Submission error:', error);
                showToast('error', 'Submission Error', error.message);
            }
        }
        async function submitWritingAnswer() {
            const userAnswer = document.getElementById('writingAnswer').value.trim();
            const wordCount = userAnswer.split(/\s+/).length;

            if (!userAnswer) {
                showToast('warning', 'Empty Essay', 'Please write your essay');
                return;
            }

            if (wordCount < 50) {
                showToast('warning', 'Too Short', `Write at least 50 words (Current: ${wordCount} words)`);
                return;
            }

            try {
                // Show loading
                showToast('info', 'Analyzing', '⏳ AI is analyzing your essay and generating feedback...');
                document.getElementById('writingAnswer').disabled = true;

                // Stop the timer
                if (writingTimerInterval) {
                    clearInterval(writingTimerInterval);
                    writingTimerInterval = null;
                }

                const response = await fetch(`${API_BASE_URL}/stage1/write-challenge/submit`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                        'Authorization': `Bearer ${localStorage.getItem('access_token')}`
                    },
                    body: JSON.stringify({
                        challenge_id: writingState.currentChallenge._id,
                        written_text: userAnswer,
                        time_taken: Math.max(0, Math.round(((writingState.currentChallenge.duration || 10) * 60) - writingState.timeLeft))
                    })
                });

                if (response.ok) {
                    const result = await response.json();
                    console.log('✅ Writing Evaluation Result:', result);

                    // Show feedback screen
                    document.getElementById('writingChallengeContent').style.display = 'none';
                    const feedbackDiv = document.getElementById('writingFeedback');
                    const feedbackTitle = document.getElementById('writingFeedbackTitle');
                    const feedbackMessage = document.getElementById('writingFeedbackMessage');

                    feedbackTitle.innerHTML = `<i class="fas fa-check-circle" style="color: #10b981;"></i> AI Analysis Complete!`;

                    // Build rich feedback HTML
                    let feedbackHTML = `
                        <div style="background: rgba(30, 41, 59, 0.5); padding: 25px; border-radius: 15px; margin-top: 20px; border: 1px solid var(--glass-border);">
                            <div style="display: flex; justify-content: space-around; margin-bottom: 25px; text-align: center;">
                                <div>
                                    <div style="font-size: 24px; font-weight: 700; color: var(--accent);">${result.score}%</div>
                                    <div style="font-size: 12px; color: var(--text-muted);">OVERALL</div>
                                </div>
                                <div style="border-left: 1px solid var(--glass-border); padding-left: 20px;">
                                    <div style="font-size: 24px; font-weight: 700; color: var(--primary);">${result.word_count}</div>
                                    <div style="font-size: 12px; color: var(--text-muted);">WORDS</div>
                                </div>
                                <div style="border-left: 1px solid var(--glass-border); padding-left: 20px;">
                                    <div style="font-size: 24px; font-weight: 700; color: #10b981;">+${result.credits_earned}</div>
                                    <div style="font-size: 12px; color: var(--text-muted);">CREDITS</div>
                                </div>
                            </div>
                            
                            <div style="margin-bottom: 20px;">
                                <h4 style="color: var(--text); margin-bottom: 8px;"><i class="fas fa-chalkboard-teacher" style="color: #6366f1;"></i> Teacher's Feedback:</h4>
                                <p style="color: var(--accent); font-style: italic; font-size: 14px; line-height: 1.6; background: rgba(99, 102, 241, 0.05); padding: 15px; border-radius: 10px; border-left: 3px solid #6366f1;">
                                    "${result.praise || 'You did a great job! Keep it up!'}"
                                </p>
                            </div>

                            <div style="margin-bottom: 20px;">
                                <h4 style="color: var(--text); margin-bottom: 8px;"><i class="fas fa-language" style="color: var(--primary);"></i> Tamil Explanation:</h4>
                                <p style="color: var(--text); font-size: 15px; line-height: 1.6; background: rgba(99, 102, 241, 0.05); padding: 15px; border-radius: 10px; border-left: 3px solid var(--primary);">
                                    ${result.tamil_feedback || 'மிக அருமையான முயற்சி!'}
                                </p>
                            </div>

                            <div style="margin-bottom: 20px;">
                                <h4 style="color: var(--text); margin-bottom: 8px;"><i class="fas fa-lightbulb" style="color: #f59e0b;"></i> Suggestions:</h4>
                                <ul style="color: var(--text-muted); font-size: 14px; line-height: 1.6; padding-left: 20px;">
                                    ${(result.suggestions || []).map(s => `<li>${s}</li>`).join('') || '<li>Keep practicing to improve even more!</li>'}
                                </ul>
                            </div>
                        </div>
                    `;

                    feedbackMessage.innerHTML = feedbackHTML;
                    feedbackDiv.style.display = 'block';

                    // Play audio if available
                    if (result.audio_url) {
                        try {
                            const audio = new Audio(result.audio_url);
                            audio.play();
                            showToast('info', 'Voice Feedback', 'Akka is giving you feedback... 🎧');
                        } catch (e) {
                            console.warn('Audio playback failed:', e);
                        }
                    }

                    showToast('success', 'Analysis Complete', `Summary: ${result.score}% Score. Excellent effort!`);
                } else {
                    const errorData = await response.json();
                    let errorMessage = 'Failed to analyze essay';
                    if (errorData.detail) {
                        errorMessage = typeof errorData.detail === 'string' ? errorData.detail : JSON.stringify(errorData.detail);
                    }
                    throw new Error(errorMessage);
                }
            } catch (error) {
                console.error('❌ Submission Error:', error);
                showToast('error', 'Submission Error', error.message || 'Could not evaluate your essay. Please try again.');
                document.getElementById('writingAnswer').disabled = false;

                // Restart timer if it crashed
                if (!writingTimerInterval && writingState.timeLeft > 0) {
                    startWritingTimer();
                }
            }
        }
        async function submitSpeakingRecording() {

                showToast('info', 'Analyzing', '⏳ AI is analyzing your pronunciation and grammar...');
                document.getElementById('speakingSubmitBtn').disabled = true;

                // Create audio blob
                const audioBlob = new Blob(speakingAudioChunks, { type: 'audio/webm' });
                const formData = new FormData();
                formData.append('audio', audioBlob, 'recording.webm');
                formData.append('question', speakingState.currentChallenge?.instructions || 'Answer the question');

                // Stop the timer
                if (speakingTimerInterval) {
                    clearInterval(speakingTimerInterval);
                    speakingTimerInterval = null;
