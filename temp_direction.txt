        async function startDirectionFollower() {
            try {
                document.getElementById('listeningSourceScreen').style.display = 'none';
                document.getElementById('directionFollowerScreen').style.display = 'block';
                document.getElementById('directionMapGrid').innerHTML = '<i class="fas fa-spinner fa-spin"></i>';
                
                const response = await fetch(`${API_BASE_URL}/communication/generate-listening-direction`, {
                    method: 'POST',
                    headers: { 'Authorization': `Bearer ${localStorage.getItem('access_token')}` }
                });
                
                if (response.ok) {
                    const data = await response.json();
                    directionState.currentChallenge = data;
                    directionState.selectedLocation = null;
                    document.getElementById('directionSelectedLocation').textContent = '';
                    
                    const grid = document.getElementById('directionMapGrid');
                    grid.innerHTML = '';
                    data.map_locations.forEach((loc, idx) => {
                        const div = document.createElement('div');
                        div.className = 'map-location';
                        div.innerHTML = `<i class="fas ${loc.icon || 'fa-building'}"></i><span style="font-size: 10px; display: block;">${loc.name}</span>`;
                        div.style.cssText = 'background: rgba(255,255,255,0.1); border-radius: 8px; display: flex; flex-direction: column; align-items: center; justify-content: center; cursor: pointer; transition: all 0.2s; padding: 5px; text-align: center;';
                        div.onclick = () => {
                            document.querySelectorAll('.map-location').forEach(el => el.style.background = 'rgba(255,255,255,0.1)');
                            div.style.background = 'rgba(99, 102, 241, 0.4)';
                            div.style.boxShadow = '0 0 10px var(--primary)';
                            directionState.selectedLocation = loc.name;
                            document.getElementById('directionSelectedLocation').textContent = `Selected: ${loc.name}`;
                            evaluateDirection(loc.name);
                        };
                        grid.appendChild(div);
                    });
                    
                    // Start timer
                    directionState.timeLeft = 180;
                    if (directionTimerInterval) clearInterval(directionTimerInterval);
                    directionTimerInterval = setInterval(() => {
                        directionState.timeLeft--;
                        const mins = Math.floor(directionState.timeLeft / 60);
                        const secs = directionState.timeLeft % 60;
                        document.getElementById('directionTimer').textContent = `${String(mins).padStart(2, '0')}:${String(secs).padStart(2, '0')}`;
                    }, 1000);
                    
                    showToast('info', 'Listen Carefully', 'Follow the directions to find the destination');
                }
            } catch (e) {
                showToast('error', 'Error', 'Failed to load Direction Follower');
                backToListeningMenu();
            }
        }
