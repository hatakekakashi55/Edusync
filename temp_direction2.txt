        async function evaluateDirection(selectedName) {
            try {
                showToast('info', 'Checking', 'Verifying location...');
                const response = await fetch(`${API_BASE_URL}/communication/evaluate-listening-direction`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${localStorage.getItem('access_token')}` },
                    body: JSON.stringify({
                        selected_location: selectedName,
                        correct_location: directionState.currentChallenge.correct_location,
                        challenge_id: directionState.currentChallenge.id,
                        difficulty: directionState.currentChallenge.difficulty
                    })
                });
                
                if (response.ok) {
                    const result = await response.json();
                    if (directionTimerInterval) clearInterval(directionTimerInterval);
                    
                    setTimeout(() => {
                        document.getElementById('directionFollowerScreen').style.display = 'none';
                        const feedbackDiv = document.getElementById('listeningFeedback');
                        document.getElementById('feedbackTitle').innerHTML = result.is_correct ? '<i class="fas fa-check-circle" style="color: #10b981;"></i> Correct Destination!' : '<i class="fas fa-times-circle" style="color: #ef4444;"></i> Wrong Location';
                        document.getElementById('feedbackMessage').innerHTML = `
                            <p>${result.feedback}</p>
                            <p style="margin-top: 10px; color: var(--text-muted); font-style: italic;">${result.tamil_feedback || ''}</p>
                            <div style="margin-top: 15px; padding: 10px; background: rgba(255,255,255,0.05); border-radius: 8px;">
                                <strong>Transcript:</strong> <br>
                                <span style="font-size: 13px;">${directionState.currentChallenge.directions_text}</span>
                            </div>
                        `;
                        feedbackDiv.style.display = 'block';
                        loadUserData();
                    }, 500);
                }
            } catch (e) {
                showToast('error', 'Error', 'Evaluation failed');
            }
        }
